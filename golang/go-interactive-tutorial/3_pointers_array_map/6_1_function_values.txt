/*
We may pass functions as arguments.

Functions are values too. They can be passed around just like other values. Function values may be used as function arguments and return values. Me: Notice that neither of the type nor lambda like has name, because the name is given to the argument of course!
*/
package main

import (
	"fmt"
	"math"
)


func compute(fn func(float64, float64) float64) float64 {
	return fn(3, 4)
}

func main() {
	hypot := func(x, y float64) float64 {
		return math.Sqrt(x*x + y*y)
	}
	fmt.Println(hypot(5, 12))

	fmt.Println(compute(hypot))
	fmt.Println(compute(math.Pow))
}
