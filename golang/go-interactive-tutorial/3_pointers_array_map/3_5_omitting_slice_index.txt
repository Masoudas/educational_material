package main

import "fmt"

func main() {
	s := []int{2, 3, 5, 7, 11, 13}

	s = s[1:4]
	fmt.Println(s)

    // Eliminating the start index. It's zero.
	s = s[:2]
	fmt.Println(s)

    // Eliminating the last index. It's end item.
	s = s[1:]
	fmt.Println(s)
    
    // From start to end.
	s = s[:]

}
